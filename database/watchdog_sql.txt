create database watchdog;
use watchdog;

create table location (
id int not null auto_increment primary key,
seq int(10) unsigned,
dog_id decimal(4,0) unsigned,
unit decimal(2,0) unsigned,
type char(1),
latitude float(10,6),
longitude float(10,6),
ip char(15),
time TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

GRANT ALL PRIVILEGES ON watchdog.* TO 'me'@'localhost';
FLUSH PRIVILEGES;
